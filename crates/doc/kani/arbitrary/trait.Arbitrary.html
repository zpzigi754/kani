where
    Self: Sized,
{
    // Required method
    fn any() -> Self;

    // Provided method
    fn any_array<const MAX_ARRAY_LENGTH: usize>() -> [Self; MAX_ARRAY_LENGTH]
       where [(); { _ }]: { ... }
}
Expand description

This trait should be used to generate symbolic variables that represent any valid value of its type.
